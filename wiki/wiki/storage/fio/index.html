    /*
 * A header structure associated with each checksummed data block. It is

 * data.

	uint32_t len;		// 4:
	uint64_t rand_seed;	// 8:
	uint64_t offset;	// 16:
	uint32_t time_sec;	// 24:
	uint32_t time_nsec;	// 28:
	uint16_t thread;	// 32:
	uint16_t numberio;	// 34:
	uint32_t crc32;		// 36:
};
