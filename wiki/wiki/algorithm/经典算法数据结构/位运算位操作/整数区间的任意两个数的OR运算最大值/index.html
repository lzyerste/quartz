    def to_bin(x):
    return bin(int(x))[2:]

def solve(a, b):
    if a == b: return a
    a, b = to_bin(a), to_bin(b)
    MAX = max(len(a), len(b))
    a = "0" * (MAX - len(a)) + a
    b = "0" * (MAX - len(b)) + b
    for i in range(MAX):
        if a[i] != b[i]:
            break
    res = a[:i] + "1" * (MAX - i)
    return int(res, base=2)

    assert solve(4611686018427387973, 4819578321603258783) == 4899916394579099647
