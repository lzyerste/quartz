    1

    class Solution:
    def rangeBitwiseAnd(self, m: int, n: int) -> int:
        def bitands2(a, b):
            c = a ^ b
            c |= c >> 1
            c |= c >> 2
            c |= c >> 4
            c |= c >> 8
            c |= c >> 16
            return a & ~c

        return bitands2(m, n)
