    class Solution:
    def findPeakElement(self, nums: List[int]) -> int:
        L, R = 0, len(nums) - 1
        while L <= R:
            M = (L + R) // 2
            left_ok = (M == 0) or (nums[M] > nums[M-1])
            right_ok = (M == len(nums) - 1) or (nums[M] > nums[M+1])
            if left_ok and right_ok: return M
            if left_ok: L = M + 1
            else: R = M - 1
