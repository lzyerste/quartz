    class Solution:
    def majorityElement(self, nums: List[int]) -> int:
        count = 0
        n = None
        for v in nums:
            if count == 0:
                n = v
                count = 1
            elif v == n:
                count += 1
            elif v != n:
                count -= 1
        return n
