    class Solution(object):
    def detectCycle(self, head):
        """
        :type head: ListNode
        :rtype: ListNode
        """
        visited = {}
        curr = head
        while curr:
            if curr in visited:
                return curr
            visited[curr] = True
            curr = curr.next
        return None

    class Solution(object):
    def detectCycle(self, head):
        """
        :type head: ListNode
        :rtype: ListNode
        """
        if not head or not head.next:
            return None
        fast = slow = head
        is_cycle = False
        while fast and fast.next:
            fast = fast.next.next
            slow = slow.next
            if fast == slow:
                is_cycle = True
                break
        if not is_cycle:
            return None
        finder = head
        while finder != slow:
            finder = finder.next
            slow = slow.next
        return finder
