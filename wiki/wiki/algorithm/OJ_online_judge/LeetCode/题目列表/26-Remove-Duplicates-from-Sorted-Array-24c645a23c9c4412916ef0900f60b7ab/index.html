    class Solution:
    def removeDuplicates(self, nums: List[int]) -> int:
        if not nums: return 0
        
        i = 0
        for d in nums:
            if d != nums[i]:
                i += 1
                nums[i] = d
        return i + 1

    class Solution:
    def removeDuplicates(self, nums: List[int]) -> int:
        i, pre = 0, None
        for d in nums:
            if d == pre: continue
            pre = d
            nums[i] = d
            i += 1
        return i
