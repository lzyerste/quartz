        dummy = NestedInteger()

        top = stack[-1]
        i = 0
        n = len(s)
        while i < n:

    class Solution:
    def deserialize(self, s: str) -> NestedInteger:
        stack = []
        i, n = 0, len(s)
        while i < n:
            c = s[i]
            if c == '[':
                stack.append(('[', NestedInteger()))
                i += 1
            elif c == ']':
                for j in range(len(stack) - 1, -1, -1):
                    t, x = stack[j]
                    if t == '[': break
                for t, y in stack[j+1:]:
                    x.add(y)
                del stack[j:]
                stack.append(('', x))  # not List any more
                i += 1
            elif c == ',':
                i += 1
            else:  # integer
                j = i + 1
                while j < n and s[j].isdigit():
                    j += 1
                x = int(s[i:j])
                stack.append(('', NestedInteger(x)))
                i = j
        return stack[0][1]

    class Solution:
    def deserialize(self, s):
        """
        :type s: str
        :rtype: NestedInteger
        """
        # each [ and int respond to a NestedInteger
        stk_s = list()  # string stack
        stk_i = list()  # Nested stack
        
        i = 0
        while i < len(s):
            c = s[i]
            if c == "[":
                stk_s.append(c)
                stk_i.append(NestedInteger())  # list integer
            elif c == "]":
                j = len(stk_i) - 1
                while stk_s[j] != "[":
                    j -= 1
                for k in range(j + 1, len(stk_i)):
                    stk_i[j].add(stk_i[k])
                **stk_s[j] = ""  # list to an element**
                del stk_i[j + 1:]
                del stk_s[j + 1:]
            elif c == ",":
                pass
            else:
                # int
                j = i + 1
                while j < len(s) and s[j].isdigit():
                    j += 1
                aint = int(s[i:j])
                stk_s.append(aint)
                stk_i.append(NestedInteger(aint))
                i = j
                continue
            i += 1
        return stk_i.pop()
