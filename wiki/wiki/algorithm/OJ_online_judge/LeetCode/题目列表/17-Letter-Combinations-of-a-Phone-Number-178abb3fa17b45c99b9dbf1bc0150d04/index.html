    def letterCombinations(self, digits: str) -> List[str]:

            "2": "abc",
            "3": "def",
            "4": "ghi",
            "5": "jkl",
            "6": "mno",
            "7": "pqrs",
            "8": "tuv",
            "9": "wxyz"
        }
        
        def dfs(digits, G):

    class Solution:
    def letterCombinations(self, digits: str) -> List[str]:
        G = {
            "2": "abc",
            "3": "def",
            "4": "ghi",
            "5": "jkl",
            "6": "mno",
            "7": "pqrs",
            "8": "tuv",
            "9": "wxyz"
        }
        
        def bp(G, digits, index, path, res):
            if index == len(digits):
                res.append(path)
                return
            for c in G[digits[index]]:
                bp(G, digits, index + 1, path + c, res)
        
        if digits == "": return []
        res = []
        bp(G, digits, 0, "", res)
        return res
