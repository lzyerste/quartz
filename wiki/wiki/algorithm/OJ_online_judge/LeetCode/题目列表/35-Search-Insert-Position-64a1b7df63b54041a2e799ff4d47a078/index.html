    class Solution:
    def searchInsert(self, nums: List[int], target: int) -> int:
        def bsearch(nums, L, R, T):
            while L <= R:
                M = (L + R) // 2
                if nums[M] < T:
                    L = M + 1
                elif nums[M] > T:
                    R = M - 1
                else:  # found
                    return M
            return L  # insert position, target < A[L]
        
        return bsearch(nums, 0, len(nums) - 1, target)
