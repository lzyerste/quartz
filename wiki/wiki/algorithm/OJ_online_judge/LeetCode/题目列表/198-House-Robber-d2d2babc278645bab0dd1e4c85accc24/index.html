    def dp(self, G, nums, n):
        if n in G: return G[n]
        if n < 0: return 0
        G[n] = max(self.dp(G, nums, n - 2) + nums[n], self.dp(G, nums, n - 1))
        return G[n]

    def rob(self, nums: List[int]) -> int:
        first = second = current = 0
        for n in nums:
            current = max(n + first, second)
            first, second = second, current
        return current
