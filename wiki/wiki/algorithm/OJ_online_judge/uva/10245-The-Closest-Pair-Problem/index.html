    def distance(p1, p2):
    return ((p1[0] - p2[0]) ** 2 + (p1[1] - p2[1]) ** 2) ** 0.5

def solve_brute_force(X, Y):
    n = len(X)
    d = float("inf")
    for i in range(n - 1):
        p1 = X[i]
        for j in range(i + 1, n):
            p2 = X[j]
            d = min(d, distance(p1, p2))
    return d

def solve_divide_conquer(X, Y):
