    def main():
    N = 10 ** 5
    n = read_int()
    C = Counter(read_ints())
    G = [0 for _ in range(N + 1)]
    H = [0 for _ in range(N + 1)]
    for k in range(1, N + 1):
        G[k] = H[k-1] + C[k] * k
        H[k] = max(G[k-1], H[k-1])
    print(max(G[N], H[N]))
