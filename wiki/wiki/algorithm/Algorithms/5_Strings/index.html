def search(pat, txt):

    N = len(txt)
    for i in range(N - M + 1):  # ä¿è¯txtä¸­æœ‰Mä¸ªå­—ç¬¦

def search(pat, txt):
    M = len(pat)
    N = len(txt)
    i = j = 0
    while i < N and j < M:
        if txt[i] == pat[j]:
            j += 1
        else:
            i -= j  # iå›åˆ°æ¯”è¾ƒçš„èµ·ç‚¹
