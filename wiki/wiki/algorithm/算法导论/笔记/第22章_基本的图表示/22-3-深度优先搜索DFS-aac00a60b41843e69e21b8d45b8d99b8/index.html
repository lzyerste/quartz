    WHITE = 0
GRAY = 1
BLACK = 2

def timestamp():
    t = 0
    while True:
        t += 1
        yield t

def dfs(G, C, P, B, E):
    """
    G: graph
    C: color
    P: parenthood
    B: begin timestamp
    E: end timestamp
    """
    time = timestamp()

    def dfs_visit(G, u):
        assert(C[u] == WHITE)
        B[u] = next(time)
        C[u] = GRAY
        for v in G[u]:
            if C[v] == WHITE:
                P[v] = u
                dfs_visit(G, v)
        C[u] = BLACK
        E[u] = next(time)

    for u in G:
        C[u] = WHITE
        P[u] = None
    for u in G:
        if C[u] == WHITE:
            dfs_visit(G, u)
