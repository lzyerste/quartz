    def topological_sort_2(G):
    C = {}
    P = {}
    topo = []
    has_cycle = False

    def dfs(G, u):
        C[u] = GRAY
        for v in G[u]:
            if C[v] == GRAY: has_cycle = True
            if C[v] == WHITE:
                P[v] = u
                dfs(G, v)
        C[u] = BLACK
        topo.append(u)

    for u in G:
        C[u] = WHITE
        P[u] = None
    for u in G:
        if C[u] == WHITE:
            dfs(G, u)
    return topo[::-1]
