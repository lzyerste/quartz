    def dijkstra(G, s, D, P):
    # 24.3, page 383
    import heapq

    initialize_single_source(G, s, D, P)
    vis = set()
    Q = [(D[u], u) for u in G]
    heapq.heapify(Q)  # one node may have multiple distances
    while Q:
        d, u = heapq.heappop(Q)
        if u in vis: continue
        vis.add(u)  # d == D[u] here
        for v in G[u]:
            if D[u] + G[u][v] < D[v]:  # relax
                D[v] = D[u] + G[u][v]
                P[v] = u
                heapq.heappush(Q, (D[v], v))
